Injected service — PkToastrService. Requires <pk-toastr /> in root shell. 

import { PkToastrService } from '../shares/pk-toastr/pk-toastr.service';

private readonly toastr = inject(PkToastrService);

this.toastr.success('Saved!', 'Success');
this.toastr.error('Failed to save.', 'Error');
this.toastr.warning('Check your input.', 'Warning', 5000);
this.toastr.info('Update available.', 'Info');

<pk-tabs [tabs]="..." [(activeId)]="..." variant="line|pill">
Use the *pkTabPanel="'id'" directive to define each tab's content.

variant="line" (default)

Overview — Summary of current status.
Active tab: {{ activeLineTab() }}
Details — Full details and configuration.
Logs — Activity and event log.

variant="pill"

Daily view — {{ activePillTab() }}
Weekly view
Monthly view
 
import { PkTabs, PkTabPanel } from '../shares/pk-tabs';
import type { PkTab } from '../shares/pk-tabs';

// In component imports: [PkTabs, PkTabPanel]

tabs: PkTab[] = [
  { id: 'overview', label: 'Overview' },
  { id: 'details',  label: 'Details'  },
  { id: 'disabled', label: 'Disabled', disabled: true },
];
activeTab = signal('overview');

<pk-tabs [tabs]="tabs" [(activeId)]="activeTab">
  <div *pkTabPanel="'overview'">Overview content</div>
  <div *pkTabPanel="'details'">Details content</div>
</pk-tabs>

<!-- Pill style -->
<pk-tabs [tabs]="tabs" [(activeId)]="activeTab" variant="pill"> ... </pk-tabs>
